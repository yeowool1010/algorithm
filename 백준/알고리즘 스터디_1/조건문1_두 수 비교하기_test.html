const readline = require('readline');

const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout,
});

rl.on('line', function (line) {
    const input = line.split(' ');

    const A = Number(input[0]);
    const B = Number(input[1]);

   if(A > B) {
        console.log(">") 
    } else if(A < B) {
        console.log("<")
    } else if(A === B) {
        console.log("==")
    }

    rl.close();
}).on('close', function () {
    process.exit();
});
